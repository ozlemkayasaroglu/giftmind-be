# Netlify konfigürasyonu
# Bu dosyayı frontend projesinin root'una netlify.toml adıyla kaydet

[build]
  # Build komutu (örnek: React için)
  command = "npm run build"
  # Build output klasörü
  publish = "dist"
  # veya React için: publish = "build"
